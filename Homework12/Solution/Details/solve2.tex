\begin{solution}
    \textbf{约定：}
    $P(x)$：$x$是有理数；
    $Q(x)$：$x$是实数；
    $R(x)$：$x$是无理数；
    $S(x)$：$x$是虚数.

    \textbf{已知：}
    \begin{align*}
        \forall x P(x) &\rightarrow Q(x)\\
        \forall x R(x) &\rightarrow Q(x)\\
        \forall x S(x) &\rightarrow \neg Q(x)
    \end{align*}

    \textbf{求证：}
    $$\forall x S(x) \rightarrow \neg P(x) \wedge \neg R(x)$$

    \begin{proof}~\\
